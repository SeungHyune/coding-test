  for (let j = 0; j < M; j++) {
    const checkNum = arr[i][j];

    let dx = i;
    let dy = j;
    let k = 1;
    while (true) {
      if (dx + k >= N || dy + k >= M) break;

      const topRight = arr[dx][dy + k];
      const bottomLeft = arr[dx + k][dy];
      const bottomRight = arr[dx + k][dy + k];

      if (
        checkNum === topRight &&
        checkNum === bottomLeft &&
        checkNum === bottomRight &&
        answer < (k + 1) * (k + 1)
      ) {
        answer = (k + 1) * (k + 1);
      }

      k++;
    }
  }
}

console.log(answer);
