let input = fs.readFileSync('/dev/stdin').toString().trim().split('\n');

const N = Number(input[0])

function solution(N) {
    const number = [1, 5, 10, 50]
    const result = new Set();
    function dfs(L, num) {
        if(L === N) {
            result.add(num);
        } else {
            for(let i = 0; i < 4 ; i++) {
                dfs(L+1, num + number[i]);
            }
        }
    }
    dfs(0, 0);
    
    return result.size;
}

console.log(solution(N))
